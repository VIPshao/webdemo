 91      	$.extend($["om"]["omGrid"].prototype , {
 92      		/** <span style='color:red'>(作用于行编辑插件)。</span>取消编辑状态。如果当前某行正处于编辑状态，取消此次的行编辑，相当于点击了行编辑条的“取消”按钮。
 93      		 * @function

112         	 * @function
113         	 * @name omGrid#cancleChanges
114         	 * @returns jQuery对象
115         	 * @example
116         	 * $(".selector").omGrid("cancleChanges");

129         	 * @function
130         	 * @param index 行索引，从0开始 
131         	 * @name omGrid#editRow
132         	 * @returns jQuery对象
133         	 * @example
134         	 * $(".selector").omGrid("editRow" , 1);

143         	/** <span style='color:red'>(作用于行编辑插件)。</span>删除某一行，如果该行是新添加的并未保存到后台，则进行物理删除；如果该行是原本就存在的，则只是隐藏并进行标记,当调用了saveChanges后才进行物理删除。
144         	 * @function
145         	 * @param index 行索引，从0开始 
146         	 * @name omGrid#deleteRow
147         	 * @returns jQuery对象
148         	 * @example
149         	 * $(".selector").omGrid("deleteRow" , 0);
150         	 */
151         	deleteRow : function(index){
152         		if(!_isEditable()){
153         			return ;
154         		}
155         		var $tr = self._getTrs().eq(index);
156         		self.cancleEdit();
157         		var rowId = _getRowId($tr);
158         		if($tr.attr("_insert")){
159         			delete self._changeData["insert"][rowId];
160         			$tr.remove();
161         		}else{
162         			self._changeData["delete"][rowId] = self._rowIdDataMap[rowId];
163         			$tr.attr("_delete", "true");
164         			$tr.hide();
165         		}
166         	},

168         	/** <span style='color:red'>(作用于行编辑插件)。</span>获取所有未保存的修改。如果没有指定type,返回的是所有的修改，格式为: {update:[],insert:[],delete:[]}，如果指定了参数，如
169         	 *  指定了"update"，则返回 [{},{}]
170         	 * @function
171         	 * @param type 可选值为："insert","update","delete" 
172         	 * @name omGrid#getChanges
173         	 * @returns 若指定了类型，返回[]，否则返回{update:[],insert:[],delete:[]}
174         	 * @example
175         	 * $(".selector").omGrid("getChanges" , "update");
176         	 */
177         	getChanges : function(type){
178         		var data = {"update":[] , "insert":[] , "delete":[]},
179         			reqType = type? type : "update",
180         			changeData = self._changeData,
181         			i;
182         			
183         		if(reqType === "update"){
184         			var uData = changeData[reqType];
185         			for(i in uData){
186         				own.call(uData , i) && data[reqType].push($.extend(true , {} , self._rowIdDataMap[i] , uData[i]));
187         			}
188         			reqType = type? type : "insert";
189         		}
190         		if(reqType === "insert"){
191         			var iData = changeData[reqType];
192         			for(i in iData){
193         				own.call(iData , i) && data[reqType].push(iData[i]);
194         			}
195         			reqType = type? type : "delete";
196         		}
197         		if(reqType === "delete"){
198         			var dData = changeData[reqType];
199         			for(i in dData){
200         				own.call(dData , i) && data[reqType].push(dData[i]);
201         			}
202         		}
203         		if(type){
204         			return data[type];
205         		}else{
206         			return data;
207         		}
208         	},

210     	 	/** <span style='color:red'>(作用于行编辑插件)。</span>在指定位置动态插入一行。
211 	    	 * @function
212 	    	 * @param index 行索引，从0开始，或为"begin","end"分别表示在表格最前和最后插入行。
213 	    	 * @param rowData 插入的新行的初始值
214 	    	 * @name omGrid#insertRow
215 	    	 * @returns jQuery对象
216 	    	 * @example
217 	    	 * $(".selector").omGrid("insertRow" , "end" , {id:1 , name:"飞儿"});
218 	    	 */
219         	insertRow : function(index , rowData){
220         		if(!_isEditable()){
221         			return ;
222         		}
223         		if($.isPlainObject(index)){
224         			rowData = index;
225         			index = 0;
226         		}
227         		var $trs = self._getTrs();
228         		index = ("begin"==index || index==undefined)? 0 : ("end"==index? $trs.length : index);
229         		
230         		//处理行数据
231         		var rd = {};
232         		for(var i=0,len=colModel.length; i<len; i++){
233         			rd[colModel[i]["name"]] = "";//默认都为空值
234             	}
235         		self._changeData["insert"][self._guid] = $.extend(true , rd , rowData);
236         		
237         		//创建新行
238         		var rowValues=self._buildRowCellValues(colModel,rd,index),
239         			trContent = [],
240         			rowClasses=options.rowClasses;
241         			isRowClassesFn= (typeof rowClasses === 'function'),
242         			rowCls = isRowClassesFn? rowClasses(index,rd):rowClasses[index % rowClasses.length],
243         			tdTmp = "<td align='$' abbr='$' class='grid-cell-dirty $'><div align='$' class='$' style='width:$px'>$</div></td>";//td模板
244         			    			
245         		trContent.push("<tr class='om-grid-row " + rowCls + "' _grid_row_id="+(self._guid++)+" _insert='true'>");
246         		$(gridHeaderCols).each(function(i){
247                     var axis = $(this).attr('axis'),
248                     	wrap=false,
249                     	html,
250                     	cols,
251                     	j;
252                     if(axis == 'indexCol'){
253                         html="<a class='om-icon'>新行</a>";
254                     }else if(axis == 'checkboxCol'){
255                         html = '<span class="checkbox"/>';
256                     }else if(axis.substring(0,3)=='col'){
257                         var colIndex=axis.substring(3);
258                         html=rowValues[colIndex];
259                         if(colModel[colIndex].wrap){
260 							wrap=true;
261 						} 
262                     }else{
263                         html='';
264                     }
265                     cols = [this.align , this.abbr , axis , this.align , wrap?'wrap':'', $('div',$(this)).width() , html];
266                     j=0;
267                     trContent.push(tdTmp.replace(/\$/g , function(){
268                     	return cols[j++];
269                     }));
270                 });
271                 trContent.push("</tr>");
272         		
273         		var $tr = $(trContent.join(" "));
274         		index==0? $tr.prependTo($elem.find(">tbody")) : $trs.eq(index-1).after($tr);
275         		
276         		self.editRow(index);
277         	},

279         	/** <span style='color:red'>(作用于行编辑插件)。</span>保存客户端数据。注意，此方法不会提交请求到后台，而是会认为所有的数据改变都已经成功提交到后台去了，所以它会清除所有脏数据标志。
280         	 * 一般情况下，在您自己的保存方法事件回调中，调用getChanges方法获取当前所有的改变，并自己提交到后台，然后在成功回调方法中再调用本方法。
281         	 * 一旦调用本方法后，cancleChanges方法是无法对调用此方法前的所有改变起作用的。
282 	    	 * @function
283 	    	 * @name omGrid#saveChanges
284 	    	 * @returns jQuery对象
285 	    	 * @example
286 	    	 * $(".selector").omGrid("saveChanges");
287 	    	 */
288         	saveChanges : function(){
289     			if(!_isEditable() || !_hasChange()){
290     				return ;
291     			}
292     			var changeData = self._changeData,
293     				rowsData = self.pageData.data.rows,
294     				uData = changeData["update"],
295     				dData = changeData["delete"],
296     				$trs = self.element.find("tr.om-grid-row"),
297     				newRowsData = [];
298     			for(var i in uData){
299     				if(own.call(uData , i)){
300     					$.extend(true , self._rowIdDataMap[i] , uData[i]);
301     				}
302     			}
303     			$trs.each(function(index , tr){
304     				var $tr = $(tr);
305     				if($tr.attr("_delete")){
306     					$tr.remove();
307     				}else{
308     					newRowsData.push(_getRowData(tr));
309     				}
310     			});
311     			self.pageData.data.rows = newRowsData;
312     			self._changeData = {"update":{},"insert":{},"delete":{}};//清除缓存
313     			_resetForm();//重置编辑表单，清除错误信息
314     			self.refresh();//重用dataSource中的数据进行刷新
315         	},
316         	/**
317         	 * (覆盖)获取最新的所有数据
318         	 */
319         	getData : function(){
320         		var result = this.pageData.data,
321     				$trs = self._getTrs();
322     				
323         		if(_isEditable() && _hasChange()){
324         			result = {total:result.total};
325         			result.rows = [];
326     				$trs.each(function(index , tr){
327 	    				result.rows.push(self._getRowData(index));
328     				});
329     			}
330     			return result;
331         	},
332         	/**
333         	 * (覆盖)获取最新的行数据。由于可以插入新的数据行，所以此方法要进行重写。
334         	 */
335         	_getRowData : function(index){
336         		var $tr = this._getTrs().eq(index),
337         			rowId = _getRowId($tr),
338         			rowData;
339         		if($tr.attr("_insert")){
340         			rowData = self._changeData.insert[rowId];
341         		}else{
342         			var origRowData = self._rowIdDataMap[rowId],
343         				uData = self._changeData["update"];
344         			rowData = origRowData;
345         			if(uData[rowId]){
346         				rowData = $.extend(true , {} , origRowData , uData[rowId]);
347         			}
348         		}
349         		return rowData;
350         	}

353         function editRenderer(){
354         	var $tr = $(this),
355         	$editRow,
356         	$editForm,
357         	scrollLeft;
358         	
359         	if(!_isEditable()){
360         		return ;
361         	}
362         		
363         	//如果是insertEditMode模式，那么除非该行是新增的，否则不可编辑，直接返回
364         	if(!self._allEditMode && !$tr.attr("_insert")){
365         		return ;
366         	}
367         	
368         	//当前行正处于编辑状态，直接返回
369         	if(self._triggered  
370         		&& self._editView.editing
371         		&& _getRowId($tr) == self._editView.rowId){
372 				return ;        		
373         	}
374         	
375         	_showEditView(this);
376         	$editRow = self._editView.editRow;
377         	$editForm = $editRow.find(">.grid-edit-form");
378         	scrollLeft = $elem.parent().scrollLeft();
379         	
380         	$(gridHeaderCols).each(function(index){
381             	var axis = $(this).attr('axis'),
382 					model,
383 					$cell = $tr.find("td:eq("+index+")"),
384 					name,//编辑组件input域的名字，这是校验所必需的
385 					compKey;//指editComps的key
386             	if(axis.substring(0,3)=='col'){
387                  	var colIndex=axis.substring(3);
388                  	model = colModel[colIndex];
389             	}else{
390             		if($.isEmptyObject(self._editComps)){//保证再次编辑其它行时不会重复添加 padding-left.
391             			$editRow.css("padding-left", parseInt($editRow.css("padding-left")) + $cell.outerWidth());
392             		}
393             		return ;
394             	}
395             	var editor = model.editor;
396             	if(!self._triggered){
397             		//如果列不可编辑，则默认type="text"
398                 	//列不可编辑的条件: 
399                 	//(1)colModel没有editor属性
400                 	//(2)colModel有editor属性，并且editor有editable属性，那么editable===false则不进行编辑，或者editable为函数且返回false也不进行编辑。
401     				if(!editor || 
402     	            	(editor && 
403     	            		(editor.editable===false || 
404     	                    	($.isFunction(editor.editable) && editor.editable()===false) ) ) ){
405     					var renderer = editor && editor.renderer;
406     	       			model.editor = editor = {};
407     	       			editor.type = "text";
408     	       			if(renderer){
409     	       				editor.renderer = renderer;
410     	       				editor.type = "custom";
411     	       			}
412     	       			editor.editable = false;
413     	   			}else{
414     	   				editor.type = editor.type || "text";
415     	   				editor.editable = true;
416     	   				if(editor.rules){
417     	   					self._validate = true;//只有需要检验才需要进行检验
418     	   				}
419     	   			}
420     	   			compKey = model.editor.name || model.name;
421     	   			editor.options = editor.options || {};
422     	   			self._editComps[compKey] = {};
423             	}else{
424             		compKey = model.editor.name || model.name;
425             	}
426             	editComp = self._editComps[compKey];
427 	   			lastValue = _getLastValue($tr , model) || "";
428 	   			
429 	   			//可编辑并且可校验，添加对应的出错信息显示容器
430 	   			if(!self._triggered && editor.editable && editor.rules){
431 	   				self._errorHolders[compKey] = $("<div class='errorContainer' style='display:none'></div>").appendTo($elem.parent());
432 	   			}
433 	   			var $ins = editComp.instance,
434 	   				$wrapper,
435 	   				type = editor.type;
436 	   			if(!$ins){
437 	   				$wrapper = $("<div style='position:absolute'></div>").css({left:$cell.position().left+scrollLeft,top:3}).appendTo($editForm).addClass("grid-edit-wrapper"); 
438 	   				$ins = editComp.instance = $("<input></input>").attr({"name":compKey,"id":compKey}).appendTo($wrapper);
439 	   				if("text"!=type && "custom"!=type){//实例化组件
440 	   					$ins[type](editor.options);
441 	   				}
442 	   				if("omCalendar"==type || "omCombo"==type){
443 	   					var $parent = $ins.parent();
444 	   					if("omCalendar"==type){
445 	   						$ins.val(lastValue).width($cell.outerWidth()-24);
446 	   					}
447 	   					$ins.width($cell.outerWidth(true) - ($parent.outerWidth(true) - $ins.width()));
448 	   				}else{
449 	   					if("text"==type){
450 	   						$ins.addClass("grid-edit-text");
451 	   						if(!editor.editable){
452 	   							$ins.attr("readonly" , "readonly").addClass("readonly-text");
453 	   						}
454 	   					}
455 	   					if("custom"==type){
456 	   						$wrapper.html(editor.renderer(lastValue));
457 	   					}
458 	   					$ins.width($cell.outerWidth(true) - ($ins.outerWidth(true) - $ins.width()));
459 	   				}
460 	   				editComp.model = model;
461 					editComp.type = type;
462 					editComp.id = model.name;
463 	   			}
464    				switch(type){
465    					case "omCalendar":
466    						$ins = $ins.val(lastValue).omCalendar();
467    						break;
468    					case "omNumberField":
469    						$ins.val(lastValue).trigger("blur");//进行错误处理
470    						break;
471    					case "omCombo":
472    						$ins.omCombo("value" , lastValue);
473    						break;
474    					case "text":
475    						$ins.val(lastValue);
476    						break;
477    					default:;
478    				}
479             });
480             !self._triggered && self._validate && _bindValidation();
481             	
482             self._validator && self._validator.form();//触发校验
483         	self._triggered = true;
484         }

487     function _getEditBtnPosition(){
488 		var $elem = self.element,
489 			$bDiv = $elem.parent(),
490 			ev = self._editView,
491 			$editView = ev.view,
492 			$editBtn = ev.editBtn,
493 			$editRow = ev.editRow,
494 			pos = {};
495 			
496 		pos.top = $editRow.height();
497 		if($elem.width() < $bDiv.width()){

668 	function _saveEditValue(tr){
669 		var $tr = $(tr),
670 			$editRow = self._editView.editRow,
671 			comps = self._editComps,
672 			rowId = _getRowId($tr),
673 			index = self._getTrs().index($tr),
674 			rowData = self._getRowData(index);
675 			
676 		$.each(comps , function(name , comp){
677 			var key = comp.model.name,
678 				newValue = _getCompValue($tr , comp),
679 				originalValue,
680 				html,
681 				updateRowData;
682 				
683 			if($tr.attr("_insert")){
684 				self._changeData.insert[rowId][key] = newValue;
685 			}else{
686 				originalValue = _getRowData(tr)[key];
687 				updateRowData = self._changeData.update[rowId];
688 				if(newValue == originalValue){
689 					_toggleDirtyFlag($tr , comp.model , false);
690 					updateRowData && delete updateRowData[key];
691 					$.isEmptyObject(updateRowData) && delete self._changeData.update[rowId];
692 				}else{
693 					_toggleDirtyFlag($tr , comp.model , true);
694 					updateRowData = self._changeData.update[rowId] = updateRowData || {};
695 					updateRowData[key] = newValue;
696 				}
697 			}
698 			//更新回表格
699 			if(comp.model.renderer){
700 				html = comp.model.renderer(newValue , rowData ,index);
701 			}else{
702 				html = newValue? newValue : "";
703 			}
704 			$tr.find("td[abbr='"+key+"'] >div").html(html);
705 		});
706 	}

751 				value = _getRowData($tr)[name];
752 			}
753 		}
